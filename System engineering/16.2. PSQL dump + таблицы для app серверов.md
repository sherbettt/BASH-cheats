## Подготовительные работы

### Важное правило: пользователи в PostgreSQL
В PostgreSQL используется аутентификация `peer` - системный пользователь должен соответствовать пользователю БД. Поэтому все операции нужно выполнять от пользователя `postgres`.

```bash
# Переключиться на пользователя postgres (вариант 1)
su - postgres

# Переключиться на пользователя postgres (вариант 2 - от root)
sudo -u postgres bash

# Переключиться на пользователя postgres (вариант 3 - выполнить одну команду)
sudo -u postgres <команда>
```

### Просмотр существующих баз данных
```bash
# Показать все базы данных (вариант 1 - в psql)
psql -l

# Показать все базы данных (вариант 2 - в psql интерактивно)
psql
\l

# Показать конкретную базу
psql -l | grep jira

# В psql показать информацию о конкретной базе
\l jira_restored
```

## Проверка лидера Patroni и управление кластером

### Смотрим лидера/реплику и конфигурацию
```bash
# Показать конфигурацию кластера
root@pg1:~ > patronictl -c /etc/patroni/config.yml show-config

# Показать топологию кластера (кто лидер, кто реплики)
root@pg1:~ > patronictl -c /etc/patroni/config.yml topology
+ Cluster: postgres ----------+---------+---------+-----+-----------+
| Member | Host               | Role    | State   |  TL | Lag in MB |
+--------+--------------------+---------+---------+-----+-----------+
| pg1    | 192.168.45.201:5433 | Leader  | running | 400 |           |
| pg2    | 192.168.45.202:5433 | Replica | running | 400 |         0 |
| pg3    | 192.168.45.204:5433 | Replica | running | 400 |         0 |
+--------+--------------------+---------+---------+-----+-----------+
```

### Переключение мастера (switchover)
```bash
# Выполняем switchover (переключение ролей) - переводим мастер на pg2
patronictl -c /etc/patroni/config.yml switchover --candidate pg2 --force

# Или используем более явную команду
patronictl -c /etc/patroni/config.yml switchover --master pg1 --candidate pg2

# Проверяем результат
patronictl -c /etc/patroni/config.yml topology

# Дополнительная проверка на самой ноде pg2
sudo -u postgres psql -h /var/lib/postgresql/patroni -p 5433 -c "SELECT pg_is_in_recovery();"
# Если вернет 'f' - это мастер, если 't' - реплика

# Если switchover не работает, используем failover
patronictl -c /etc/patroni/config.yml failover --candidate pg2

# Или перезапустить мастер и заставить выбрать новый
patronictl -c /etc/patroni/config.yml restart --role master
```

### Проверка роли текущей ноды
```bash
# Простой способ узнать, мастер или реплика
psql -c "SELECT pg_is_in_recovery();"
# f = false = мастер
# t = true = реплика

# Более подробная информация
psql -c "SELECT 
    pg_is_in_recovery() as is_replica,
    inet_server_addr() as server_ip,
    inet_server_port() as server_port,
    current_database() as current_db,
    version();"
```

### 1. На МАСТЕРЕ (pg1):

#### Создаем полный бэкап всех БД:

**`pg_dumpall`** - это обертка, которая:
- Делает `pg_dumpall -g` для глобальных объектов
- Для каждой БД запускает `pg_dump` отдельно
- Объединяет всё в один SQL файл

```bash
# Переходим в директорию бэкапов
cd /var/backups/postgresql/

# Создаем папку для бэкапа с мастера
mkdir -p master_full_$(date +%Y%m%d_%H%M)
cd master_full_$(date +%Y%m%d_%H%M)

# 1. Бэкап ВСЕХ БД одним файлом
pg_dumpall -h /var/lib/postgresql/patroni -p 5433 -U postgres -f all_databases.sql

# 2. Дамп всех БД со сжатием
pg_dumpall -h /var/lib/postgresql/patroni -p 5433 -U postgres | gzip -9 > all_databases.sql.gz

# Работа со сжатыми файлами:
gzip -d all_databases.sql.gz              # разархивировать
gunzip all_databases.sql.gz                # альтернативный способ
gzip -t all_databases.sql.gz               # проверка целостности
zcat all_databases.sql.gz | head -20       # посмотреть первые строки
gzip -c -d all_databases.sql.gz > restored.sql # распаковать с другим именем

# 3. Бэкап глобальных объектов (роли) - ТОЛЬКО НА МАСТЕРЕ!
pg_dumpall -h /var/lib/postgresql/patroni -p 5433 -U postgres -g -f globals.sql

# Для восстановления глобальных объектов:
psql -f globals.sql

# 4. Бэкап каждой БД отдельно в сжатом формате
for DB in logging_db media_db pbx_master2 pbxv2 pbxv2_stat jira jira_old jira_reserve; do
    echo "Бэкап БД: $DB"
    pg_dump -h /var/lib/postgresql/patroni -p 5433 -U postgres -Fc -Z 9 -f ${DB}.dump ${DB}
done

# 5. Проверяем что создалось
ls -lh *.dump

# 6. Смена прав
chown -R postgres:postgres /var/backups/postgresql/

# 7. Создание от имени postgres сразу
sudo -u postgres pg_dumpall -h /var/lib/postgresql/patroni -p 5433 -g -f globals.sql
sudo -u postgres pg_dumpall -h /var/lib/postgresql/patroni -p 5433 -f all_databases.sql
```

### 2. На РЕПЛИКАХ (pg2 и pg3):

#### На КАЖДОЙ реплике выполняем:

```bash

# Бэкап каждой БД отдельно (глобальные объекты НЕ трогаем)
for DB in logging_db media_db pbx_master2 pbxv2 pbxv2_stat jira jira_old jira_reserve; do
    echo "Бэкап БД: $DB"

# Проверяем
ls -lh *.dump
```

### 2.1. Универсальный скрипт для всех серверов
```bash
cd /var/backups/postgresql/
mkdir -p smart_backup_$(date +%Y%m%d_%H%M)
cd smart_backup_$(date +%Y%m%d_%H%M)

# Для каждой БД создаем сжатый дамп
for DB in $(sudo -u postgres psql -h /var/lib/postgresql/patroni -p 5433 -t -c "SELECT datname FROM pg_database WHERE datistemplate = false AND datname NOT IN ('postgres');"); do
    echo "Бэкап $DB"
    sudo -u postgres pg_dump -h /var/lib/postgresql/patroni -p 5433 -Fc -Z 9 -f ${DB}.dump ${DB}
done

# Только на мастере добавляем globals
if sudo -u postgres psql -h /var/lib/postgresql/patroni -p 5433 -t -c "SELECT pg_is_in_recovery();" | grep -q "f"; then
    echo "Это мастер, добавляем глобальные объекты"
    sudo -u postgres pg_dumpall -h /var/lib/postgresql/patroni -p 5433 -g -f globals.sql
fi

# Посмотреть размеры

```

### 3. Проверить размеры таблиц в самых больших БД

```bash
# Вариант 1: Через pg_stat_user_tables
psql -h /var/lib/postgresql/patroni -p 5433 -U postgres -d jira_reserve -c "
SELECT 
    schemaname,
    relname as tablename,
    pg_size_pretty(pg_total_relation_size(schemaname || '.' || relname)) as total_size,
    n_live_tup as rows
FROM pg_stat_user_tables
ORDER BY pg_total_relation_size(schemaname || '.' || relname) DESC
LIMIT 10;
"

# Вариант 2: Через pg_class и pg_namespace (более надежно)
psql -h /var/lib/postgresql/patroni -p 5433 -U postgres -d jira_reserve -c "
SELECT 
    nspname as schemaname,
    relname as tablename,
    pg_size_pretty(pg_total_relation_size(c.oid)) as total_size,
    CASE 
        WHEN c.reltuples < 0 THEN 0
        ELSE c.reltuples::bigint 
    END as estimated_rows,
    pg_size_pretty(pg_relation_size(c.oid)) as table_size,
    pg_size_pretty(pg_total_relation_size(c.oid) - pg_relation_size(c.oid)) as index_size
FROM pg_class c
LEFT JOIN pg_namespace n ON n.oid = c.relnamespace
WHERE c.relkind = 'r'
  AND nspname NOT IN ('pg_catalog', 'information_schema')
ORDER BY pg_total_relation_size(c.oid) DESC
LIMIT 15;
"

# Вариант 3: Простой и понятный
psql -h /var/lib/postgresql/patroni -p 5433 -U postgres -d jira_reserve -c "\dt+"

# Вариант 4: Только 10 самых больших таблиц с деталями
psql -h /var/lib/postgresql/patroni -p 5433 -U postgres -d jira_reserve -c "
SELECT 
    nspname as \"Схема\",
    relname as \"Таблица\",
    pg_size_pretty(pg_total_relation_size(c.oid)) as \"Общий размер\",
    pg_size_pretty(pg_relation_size(c.oid)) as \"Размер данных\",
    pg_size_pretty(pg_total_relation_size(c.oid) - pg_relation_size(c.oid)) as \"Размер индексов\",
    pg_stat_get_live_tuples(c.oid) as \"Кол-во строк\"
FROM pg_class c
LEFT JOIN pg_namespace n ON n.oid = c.relnamespace
WHERE c.relkind IN ('r', 'p')  -- обычные и партиционированные таблицы
  AND nspname NOT IN ('pg_catalog', 'information_schema')
  AND nspname !~ '^pg_toast'
ORDER BY pg_total_relation_size(c.oid) DESC
LIMIT 10;
"
```

## Расширенные методы создания дампов

### Полезные опции pg_dump

```bash
# Основные опции:
-s, --schema-only           # Только схема (без данных)
-a, --data-only             # Только данные (без схемы)
-t, --table=TABLE           # Только указанная таблица
-n, --schema=SCHEMA         # Только указанная схема
-T, --exclude-table=TABLE   # Исключить таблицу
--exclude-schema=SCHEMA     # Исключить схему
--no-owner                  # Не включать владельцев
--no-privileges             # Не включать привилегии
--no-tablespaces            # Не включать табличные пространства
--inserts                   # Использовать INSERT вместо COPY
--column-inserts            # Вставлять с именами колонок
--where="condition"         # Фильтр строк WHERE
```

SELECT 
    usename,
    passwd IS NOT NULL as has_password,
    CASE 
        WHEN passwd LIKE 'SCRAM-SHA-256%' THEN 'scram-sha-256'
        WHEN passwd LIKE 'md5%' THEN 'md5'
        ELSE 'no password or unknown'
    END as password_method,
    length(passwd) as password_hash_length
FROM pg_shadow 
WHERE usename = 'rt_pbx';

-- Все пользователи и их методы аутентификации
SELECT 
    usename,
    CASE 
        WHEN passwd LIKE 'SCRAM-SHA-256%' THEN 'scram-sha-256'
        WHEN passwd LIKE 'md5%' THEN 'md5'
        WHEN passwd IS NULL THEN 'trust/no password'
        ELSE 'other'
    END as auth_method
FROM pg_shadow 
ORDER BY usename;

## Диагностика ошибок структуры БД

### Анализ ошибок в логах Patroni

#### Типичные ошибки:

**1. Проблемы с таблицами:**
```
ОШИБКА:  отношение "418__dialer_attempt" не существует
```
Приложения ищут таблицы типа `[number]__dialer_attempt`, которых нет. Это таблицы для доменов (тенантов).

**2. Проблема с типами данных:**
```
ОШИБКА:  оператор не существует: smallint ->> unknown
```
Поле `event` в таблице `domain_event_handling` имеет тип `smallint` (число), а приложение пытается использовать его как JSON (`->>`).

### Диагностика структуры БД

```bash
# Подключись к проблемной БД
psql -h 192.168.45.201 -p 5432 -U postgres -d rt_pbx_v2_ext

# Проверим структуру проблемной таблицы
\d+ domain_event_handling

# Посмотрим тип поля event
SELECT column_name, data_type, is_nullable 
FROM information_schema.columns 
WHERE table_name = 'domain_event_handling' 
  AND column_name = 'event';

# Сколько всего таблиц?
SELECT COUNT(*) FROM information_schema.tables 
WHERE table_schema = 'public';

# Список всех таблиц
SELECT table_name FROM information_schema.tables 
WHERE table_schema = 'public' 
ORDER BY table_name;

# Проверка конкретной проблемной таблицы
SELECT * FROM domain_event_handling LIMIT 1;
```

### Исправление структуры

```sql
-- Исправить тип поля event (с integer на jsonb)
ALTER TABLE domain_event_handling 
ALTER COLUMN event TYPE jsonb 
USING event::text::jsonb;

-- Посмотреть, какие таблицы отсутствуют
SELECT 'CREATE TABLE IF NOT EXISTS ' || table_name || ' (id SERIAL PRIMARY KEY);' 
FROM information_schema.tables 
WHERE table_schema = 'public' 
  AND table_name LIKE '%__dialer_attempt'
  AND table_name NOT IN (
    SELECT table_name 
    FROM information_schema.tables 
    WHERE table_schema = 'public'
  );

-- Проверить результат
\d+ domain_event_handling
-- Или
SELECT column_name, data_type 
FROM information_schema.columns 
WHERE table_name = 'domain_event_handling' 
  AND column_name = 'event';
```

### Возможные причины проблем со структурой

1. **Команда `system_init` выполнилась не полностью**
2. **База создавалась из неполного/битого дампа**
3. **Проблема версий** - приложение ожидает одну структуру БД, а создалась другая
4. **Миграции не применились** или применились частично

### Добавление проверки структуры в Ansible

```yaml
---
- name: Verify database structure after initialization
  community.postgresql.postgresql_query:
    db: "{{ db_name }}"
    query: |
      -- Проверяем критически важные таблицы
      WITH checks AS (
        SELECT 
          'domain_event_handling.event type' as check_name,
          CASE 
            WHEN EXISTS (
              SELECT 1 
              FROM information_schema.columns 
              WHERE table_schema = 'public' 
                AND table_name = 'domain_event_handling' 
                AND column_name = 'event' 
                AND data_type IN ('json', 'jsonb')
            ) THEN 'PASS'
            ELSE 'FAIL: event should be json/jsonb'
          END as status
        UNION ALL
        SELECT 
          'total_tables_count' as check_name,
          CASE 
            WHEN COUNT(*) >= 80 THEN 'PASS'
            ELSE 'FAIL: only ' || COUNT(*) || ' tables'
          END as status
        FROM information_schema.tables 
        WHERE table_schema = 'public'
        UNION ALL
        SELECT 
          'root_user_exists' as check_name,
          CASE 
            WHEN EXISTS (
              SELECT 1 FROM public.user WHERE email = 'root@runtel.ru'
            ) THEN 'PASS'
            ELSE 'FAIL: root@runtel.ru not found'
          END as status
      )
      SELECT * FROM checks
    login_host: "{{ db_hosts[0] }}"
    login_port: "{{ db_port }}"
    login_user: "postgres"
    login_password: "{{ db_login_password }}"
  register: db_structure_check
  changed_when: false
  run_once: true
  ignore_errors: true

- name: Show database structure check results
  debug:
    msg: |
      ==== ПРОВЕРКА СТРУКТУРЫ БД ====
      {% for check in db_structure_check.query_result %}
      {{ check.check_name }}: {{ check.status }}
      {% endfor %}
  when: db_structure_check is succeeded
```
